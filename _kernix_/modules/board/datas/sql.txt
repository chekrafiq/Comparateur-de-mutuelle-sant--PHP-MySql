CREATE TABLE board (
   idboard int(10) unsigned NOT NULL auto_increment,
   title text NOT NULL,
   description text NOT NULL,
   type varchar(20) NOT NULL,
   idegroup int(10) unsigned DEFAULT '1' NOT NULL,
   nbtopic int(10) unsigned DEFAULT '0' NOT NULL,
   nbpost int(10) unsigned DEFAULT '0' NOT NULL,
   nbeleminlisttopic int(10) unsigned DEFAULT '0' NOT NULL,
   nbeleminlistreply int(10) unsigned DEFAULT '10' NOT NULL,
   nbcarinabstract int(10) unsigned DEFAULT '50' NOT NULL,
   archiveflag tinyint(3) unsigned DEFAULT '1' NOT NULL,
   interactiveflag tinyint(3) unsigned DEFAULT '0' NOT NULL,
   backendflag tinyint(3) unsigned DEFAULT '1' NOT NULL,
   openextflag tinyint(3) unsigned DEFAULT '0' NOT NULL,
   identificationlevel tinyint(3) unsigned DEFAULT '1' NOT NULL,
   moderatorflag tinyint(3) unsigned DEFAULT '0' NOT NULL,
   moderatoremail varchar(100) NOT NULL,
   lastpostdate datetime DEFAULT '0000-00-00 00:00:00' NOT NULL,
   date datetime DEFAULT '0000-00-00 00:00:00' NOT NULL,
   PRIMARY KEY (idboard),
   KEY idboard (idboard)
);

CREATE TABLE boardpost (
   idpost int(10) unsigned NOT NULL auto_increment,
   idboard int(10) unsigned DEFAULT '0' NOT NULL,
   idtheme int(10) unsigned DEFAULT '0' NOT NULL,
   idparent int(10) unsigned DEFAULT '0' NOT NULL,
   idvisitor int(11) DEFAULT '0' NOT NULL,
   title varchar(200) NOT NULL,
   content text NOT NULL,
   abstract text NOT NULL,
   link varchar(200) NOT NULL,
   nbview int(10) unsigned DEFAULT '0' NOT NULL,
   nbreply int(10) unsigned DEFAULT '0' NOT NULL,
   level tinyint(3) unsigned DEFAULT '0' NOT NULL,
   validflag tinyint(3) unsigned DEFAULT '1' NOT NULL,
   adminflag tinyint(3) unsigned DEFAULT '0' NOT NULL,
   nickname varchar(20) NOT NULL,
   email varchar(100) NOT NULL,
   url varchar(200) NOT NULL,
   idmember int(10) unsigned DEFAULT '0' NOT NULL,
   lastreplydate datetime DEFAULT '0000-00-00 00:00:00' NOT NULL,
   date datetime DEFAULT '0000-00-00 00:00:00' NOT NULL,
   PRIMARY KEY (idpost),
   KEY idpost (idpost, idboard, idtheme, idparent)
);
